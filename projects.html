AI Music Generator (Python)

Developed a machine learning model to generate music based on user inputs.

import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout

model = Sequential([
    LSTM(256, return_sequences=True, input_shape=(100, 1)),
    Dropout(0.3),
    LSTM(256, return_sequences=False),
    Dense(128, activation='relu'),
    Dense(1, activation='sigmoid')
])
model.compile(loss='binary_crossentropy', optimizer='adam')

class Object {
public:
    float x, y, velocityX, velocityY;
    void applyGravity(float gravity) {
        velocityY += gravity;
    }
};

int main() {
    Object obj;
    obj.x = 10;
    obj.y = 20;
    obj.velocityX = 5;
    obj.velocityY = 0;

    obj.applyGravity(9.8);
    std::cout << "New Y Velocity: " << obj.velocityY << std::endl;
    return 0;
}

Simple Chat Application (C++)

Developed a basic client-server chat application allowing real-time text communication.

#include 
#include 
#include 

void handleClient(asio::ip::tcp::socket& socket) {
    char data[1024];
    socket.read_some(asio::buffer(data));
    std::cout << "Client says: " << data << std::endl;
}

int main() {
    asio::io_context ioContext;
    asio::ip::tcp::acceptor acceptor(ioContext, asio::ip::tcp::endpoint(asio::ip::tcp::v4(), 1234));
    asio::ip::tcp::socket socket(ioContext);
    acceptor.accept(socket);
    std::thread(handleClient, std::ref(socket)).detach();
    return 0;
}
